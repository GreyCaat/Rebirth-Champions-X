spawn(function()
    while wait(0.5) do
        local highestnumberworldboost = 0
        local mapname = ""
        for i, v in pairs(game:GetService("Workspace").Maps:GetChildren()) do
            if string.sub(v.Name, 1, 1) == "[" then
                local name = string.gsub(v.Name, "%d", "")
                local namedone = string.gsub(name, "%[%] ", "")
    
                local numb = string.gsub(v.Name, "%a", "")
                local numb1 = string.gsub(numb, "%[", "")
                local numb2 = string.gsub(numb1, "%]", "")
                local numbdone = tonumber(numb2)
    
                if numbdone >= highestnumberworldboost then
                    highestnumberworldboost = numbdone
                    mapname = namedone
                end
            end
        end
    
        game:GetService("ReplicatedStorage").Events.WorldBoost:FireServer(mapname)
    end
end)

local loopAutoReb
local AutoReb = AutoStuff:CreateToggle("Auto Rebirth", function(state)
    
    if state == true then
        loopAutoReb = game:GetService("RunService").RenderStepped:Connect(function()
            local optionsAutoReb = {}
            for i, v in pairs(game:GetService("Players").LocalPlayer.PlayerGui.MainUI.RebirthFrame.Top.Holder.ScrollingFrame:GetChildren()) do
                if v.Name ~= "UIGridLayout" and v.Name ~= "TextLabel" then
                    if v.Visible == true then
                        local number = tonumber(v.Name)
                        table.insert(optionsAutoReb, number)
                    end
                end
            end

            local highestnumber = 0
            for i, v in pairs(optionsAutoReb) do
                if v > highestnumber then
                    highestnumber = v
                end
            end

            game:GetService("ReplicatedStorage").Events.Rebirth:FireServer(highestnumber)
        end)
    elseif state == false then
        loopAutoReb:Disconnect()
    end

end)

local AutoEquipBest = PetStuff:CreateToggle("Auto Equip Best", function(state)

    if state == true then
        repeat wait(10)
            for i, v in pairs(game:GetService("Players").LocalPlayer.Pets:GetChildren()) do
                if v.Equipped.Value == true then
                    game:GetService("ReplicatedStorage").Functions.Request:InvokeServer("Unequip",{["PetID"] = tostring(v.Name)})
                end
            end
            
            local MaxEquip = tonumber(string.sub(game:GetService("Players").LocalPlayer.PlayerGui.MainUI.PetsFrame.Top.Holder.Stats.PetEquip.Label.Text, 3, -1))
            local pets = game.Players.LocalPlayer.Pets:GetChildren()
            
            table.sort(pets, function(p1, p2)
                local A = p1.Multiplier.Value
                local B = p2.Multiplier.Value
                return A > B
            end)
            
            if #pets == 0 then
                return print("no pets")
            end
            
            for i = 1, math.min(#pets, MaxEquip) do
                local pet = pets[i]
                game:GetService("ReplicatedStorage").Functions.Request:InvokeServer("Equip",{["PetID"] = tostring(pet)})
            end
        until state == false
    elseif state == false then
    end

end)
